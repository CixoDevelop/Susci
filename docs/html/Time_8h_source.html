26static void wait(system_tick_t how_long_wait) {
27 current_process->state = TIMER_STATE;
28
29 system_tick_t actual_time = get_time();
30 system_tick_t left_to_rewind = MAX_SYSTEM_TIME - actual_time;
31
32 if (left_to_rewind < how_long_wait) {
33 current_process->scheduler_context = how_long_wait - left_to_rewind;
34 } else {
35 current_process->scheduler_context = actual_time + how_long_wait;
36 }
37}
