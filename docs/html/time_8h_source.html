24static void wait(system_tick_t how_long_wait) {
25 current_process->state = TIMER_STATE;
26
27 system_tick_t actual_time = get_time();
28 system_tick_t left_to_rewind = MAX_SYSTEM_TIME - actual_time;
29
30 if (left_to_rewind < how_long_wait) {
31 current_process->scheduler_context = how_long_wait - left_to_rewind;
32 } else {
33 current_process->scheduler_context = actual_time + how_long_wait;
34 }
35}
