95static void set_pin_direction(pin_t pin_number, pin_direction_t new_direction) {
96 uint8_t mask = create_pin_mask(pin_number);
97 volatile uint8_t *direction = create_pin_direction_pointer(pin_number);
98
99 /* This is atomic operation */
100 uint8_t sreg = SREG;
101 cli ();
102
103 if (new_direction == OUTPUT) *direction |= mask;
104 else *direction &= ~mask;
105
106 SREG = sreg;
107}
