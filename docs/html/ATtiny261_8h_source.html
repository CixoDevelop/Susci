37static inline void platform_init(void) {
38 /* Run timer in 16 bit mode */
39 TCCR0A = (1 << TCW0);
40
41 /* And F_CPU / 1024 Hz frequency */
42 TCCR0B = (1 << CS02) | (1 << CS00);
43}
