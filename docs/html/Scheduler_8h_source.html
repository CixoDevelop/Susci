87static exec_state_t create_process(
88 pid_t process_pid,
89 exec_state_t (*worker)(void*),
90 void *parameter
91) {
92 if (process_pid > MAX_PID) return PANIC_STATE;
93
94 if (process_heap[process_pid].state != EMPTY_STATE) return PANIC_STATE;
95
96 process_heap[process_pid].worker = worker;
97 process_heap[process_pid].parameter = parameter;
98 process_heap[process_pid].state = READY_STATE;
99
100 return GOOD_STATE;

140static inline exec_state_t signal_scheduler(void) {
141 if (current_signal == 0x00) return IDLE_STATE;
142
143 /*
144 * This make local copy of signal, bacause when signal ia processing, any
145 * interrupt can change global signal flag
146 */
147 signal_t signal = current_signal;
148 current_signal = 0x00;
149
150 /*
151 * This check all processes, all of it who has SIGNAL state had been
152 * checked and these who had been marked for same signal will be executed
153 */
154 current_process = process_heap + PROCESS_HEAP_SIZE;
155
156 while (current_process-- > process_heap) {
157 if (current_process->state != SIGNAL_STATE) continue;
158
159 if (current_process->scheduler_context != signal) continue;
160
161 exec_state_t return_state
162 = current_process->worker(current_process->parameter);
163
164 if (return_state == IDLE_STATE) continue;
165
166 return return_state;
167 }
168
169 return GOOD_STATE;
170}

177static inline exec_state_t standard_scheduler(void) {
178 current_process = MAX_PRIORITY_PROCESS;
179
180 while (current_process-- > MIN_PRIORITY_PROCESS) {
181 if (current_process->state != READY_STATE) continue;
182
183 exec_state_t process_state
184 = current_process->worker(current_process->parameter);
185
186 if (process_state == IDLE_STATE) continue;
187
188 return process_state;
189 }
190
191 return GOOD_STATE;
192}

200static inline exec_state_t scheduler_loop(void) {
201 exec_state_t signal = signal_scheduler();
202
203 if (signal != IDLE_STATE) return signal;
204
205 return standard_scheduler();
206}
